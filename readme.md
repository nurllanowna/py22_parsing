# ЗАПУСК ПРОЕКТА

## Инструкция

## Создать окружение
'''
python3 -m venv <название окружения>
'''
## Активировать окружение
'''
. venv/bin//activate
'''
## или
'''
source venv/bin/activate
'''
## импорт библиотек и модулей
'''
pip install -r requirements.txt
'''
## запуск процесса
'''
python3 main.py
'''
